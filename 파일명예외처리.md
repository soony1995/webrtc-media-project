func IsAllowedFileName(file string) error {
	fileName := filepath.Base(file)
	pattern := "^[a-zA-Z0-9_.-]+$" // 허용되는 문자
	match, _ := regexp.MatchString(pattern, fileName)
	if match {
		return nil
	} else {
		return fmt.Errorf("허용 되지 않음.")
	}
}
